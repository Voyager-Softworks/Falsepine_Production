59 public bool CalculateVaultData()
60 {
61 sphereCastPositions = new Vector3[sphereCastCount];
62 sphereCastHits = new RaycastHit[sphereCastCount];
63 for(int i = 0; i < sphereCastCount; i++)
64 {
65 sphereCastPositions[i] = transform.position + (transform.up * (maxVaultingHeight / sphereCastCount) * i) + (transform.up * verticalOffset);
66 Physics.SphereCast(sphereCastPositions[i], sphereCastRadius, transform.forward, out sphereCastHits[i], sphereCastDistance, layerMask);
67 }
68 if(sphereCastHits[sphereCastCount - 1].collider != null)
69 {
70 horizontalHit = Vector3.zero;
71 verticalHit = Vector3.zero;
72 vaultingHit = Vector3.zero;
73 return false;
74 }
75 Vector3 hitNormal = Vector3.zero;
76 for(int i = sphereCastCount - 2; i >= 0; i--)
77 {
78 horizontalHit = Vector3.zero;
79 if(sphereCastHits[i].collider != null)
80 {
81 horizontalHit = sphereCastHits[i].point;
82 hitNormal = sphereCastHits[i].normal;
83 break;
84 }
85 }
86 if(horizontalHit == Vector3.zero)
87 {
88 vaultingHit = Vector3.zero;
89 return false;
90 }
91 //Check if the normal is opposite to the forward vector within the max angle
92 if(Vector3.Angle(new Vector3(hitNormal.x, 0.0f, hitNormal.z), -transform.forward) > maxVaultingAngle)
93 {
94 vaultingHit = Vector3.zero;
95 horizontalHit = Vector3.zero;
96 return false;
97 }
98
99
100 RaycastHit verticalHitInfo;
101 Physics.Raycast(new Vector3(horizontalHit.x, transform.position.y + maxVaultingHeight + verticalOffset, horizontalHit.z), -transform.up, out verticalHitInfo, maxVaultingHeight, layerMask);
102 if(verticalHitInfo.collider == null)
103 {
104 vaultingHit = Vector3.zero;
105 horizontalHit = Vector3.zero;
106 return false;
107 }
108 verticalHit = verticalHitInfo.point;
109 vaultingHit.x = horizontalHit.x;
110 vaultingHit.y = verticalHit.y;
111 vaultingHit.z = horizontalHit.z;
112 if(Physics.OverlapSphere(vaultingHit + (transform.forward * (maxVaultingDepth + sphereCastRadius)), sphereCastRadius, layerMask).Length > 0)
113 {
114 vaultingHit = Vector3.zero;
115 horizontalHit = Vector3.zero;
116 verticalHit = Vector3.zero;
117 return false;
118 }
119 return true;
