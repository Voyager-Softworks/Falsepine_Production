58 _behaviour.nodeData.Where(x => !x.noLogic).All(x => x.runtimeLogic = (RuntimeBase)ScriptableObject.Instantiate(x.runtimeLogic));
59 _behaviour.nodeData.Where(x => !x.noQuery).All(x => x.query = (Query)ScriptableObject.Instantiate(x.query));
60 _behaviour.queries.Clear();
61 _behaviour.queries.AddRange(_behaviour.nodeData.Where(x => !x.noQuery).Select(x => x.query));
62 _behaviour.nodeData.Where(x => !x.noLogic).ToList().ForEach(x => x.runtimeLogic.state = NodeData.State.Idle);
63 foreach(NodeData.SerializableProperty p in inspectorProperties)
64 {
65 NodeData.SerializableProperty other = _behaviour.exposedProperties.Where(x => x.GUID == p.GUID).First();
66 other.ivalue = p.ivalue;
67 other.fvalue = p.fvalue;
68 other.svalue = p.svalue;
69 other.bvalue = p.bvalue;
70 other.ovalue = p.ovalue;
71 other.v2value = p.v2value;
72 other.v3value = p.v3value;
73 other.v4value = p.v4value;
74 other.cvalue = p.cvalue;
75 }
76 nodeTree = NodeTree.CreateFromNodeData(_behaviour.nodeData.Find(x => x.nodeType == NodeData.Type.EntryPoint), _behaviour.nodeData);;
77 nodeTree.rootLeaf.nodeData.runtimeLogic.Init(nodeTree.rootLeaf);
78 nodeTree.PropogateExposedProperties(_behaviour.exposedProperties);
79
80 _propertyMap = new Dictionary<NodeData.SerializableProperty, List<NodeData.SerializableProperty>>();
81 foreach(NodeData.SerializableProperty p in _behaviour.exposedProperties)
82 {
83 if(!_propertyMap.ContainsKey(p))
84 {
85 _propertyMap[p] = new List<NodeData.SerializableProperty>();
86
87 }
88 _propertyMap[p].AddRange(nodeTree.nodes.Where(x => !x.noLogic).SelectMany(x => x.runtimeLogic.GetPropertiesWhereParamReference(p.GUID)));
89 _propertyMap[p].AddRange(nodeTree.nodes.Where(x => !x.noQuery).SelectMany(x => x.query.GetPropertiesWhereParamReference(p.GUID)));
90
91
92 }
93 foreach(Query q in _behaviour.queries)
94 {
95 q.GetSerializableProperties().Where(x => x.output).ToList().ForEach(x =>
96 {
97 if(!_propertyMap.ContainsKey(x))
98 {
99 _propertyMap[x] = new List<NodeData.SerializableProperty>();
100
101 }
102 _propertyMap[x].AddRange(nodeTree.nodes.Where(n => !n.noQuery).SelectMany(n => n.query.GetPropertiesWhereParamReference(x.GUID)));
103 _propertyMap[x].AddRange(nodeTree.nodes.Where(n => !n.noLogic).SelectMany(n => n.runtimeLogic.GetPropertiesWhereParamReference(x.GUID)));
104 _propertyMap[x].AddRange(_behaviour.queries.SelectMany(n => n.GetPropertiesWhereParamReference(x.GUID)));
105
106 });

115 void Update()
116 {
117 if(_behaviour == null)

121 tickTimer += Time.deltaTime;
122
123 foreach (var query in _behaviour.queries)
124 {
125 query.GetNewValues(this);
126 }
127 _propertyMap?.Keys.ToList().ForEach(x =>
128 {
129 foreach(NodeData.SerializableProperty p in _propertyMap[x])
130 {
131 p.ivalue = x.ivalue;
132 p.fvalue = x.fvalue;
133 p.svalue = x.svalue;
134 p.bvalue = x.bvalue;
135 p.ovalue = x.ovalue;
136 p.v2value = x.v2value;
137 p.v3value = x.v3value;
138 p.v4value = x.v4value;
139 p.cvalue = x.cvalue;
140 }
141 });
142 if (tickTimer > tickRate)
143 {
144 tickTimer = 0f;
145 nodeTree.rootNode.Eval(this, nodeTree.rootLeaf);

155 public void SetParameter<T>(string name, T value)
156 {
157 if(_behaviour == null)
158 {
159 return;
160 }
161 foreach (var property in _behaviour.exposedProperties)
162 {
163 if (property.name == name && property.type == typeof(T))
164 {
165 switch(property.type.Name)
166 {
167 case "Int32":
168 property.ivalue = (int)(object)value;
169 break;
170 case "Single":
171 property.fvalue = (float)(object)value;
172 break;
173 case "Boolean":
174 property.bvalue = (bool)(object)value;
175 break;
176 case "String":
177 property.svalue = (string)(object)value;
178 break;
179 case "Vector2":
180 property.v2value = (Vector2)(object)value;
181 break;
182 case "Vector3":
183 property.v3value = (Vector3)(object)value;
184 break;
185 case "Vector4":
186 property.v4value = (Vector4)(object)value;
187 break;
188 case "Color":
189 property.cvalue = (Color)(object)value;
190 break;
191 default:
192 break;
193 }
194 nodeTree.PropogateExposedProperties(_behaviour.exposedProperties);
195 return;
196 }
