242 m_shootTimer = m_shootTime;
243 UpdateWaitTimer(m_shootTime);
244
245 // get all objects with healthScript in the scene
246 List<HealthScript> healthScripts = FindObjectsOfType<HealthScript>().ToList();
247 // sort by distance from the aimQuad
248 Vector3 aimQuadPos = _aimZone.transform.position;
249 healthScripts.Sort((x, y) => Vector3.Distance(x.transform.position, aimQuadPos).CompareTo(Vector3.Distance(y.transform.position, aimQuadPos)));
250
251 RHit hitInfoR = new RHit();
252
253 List<HealthScript> healthScriptsInAimZone = new List<HealthScript>();
254 foreach (HealthScript healthScript in healthScripts)
255 {
256 if (healthScript.isDead) continue;
257 // get bounds of healthScript object
258 Bounds? tempBounds = healthScript.GetComponent<Collider>()?.bounds;
259 if (tempBounds == null) continue;
260 Bounds bounds = (Bounds)tempBounds;
261
262 // split aimzone into two triangles (bl, fl, fr) and (bl, fr, br)
263 if (
264 BoundsInTriangle(bounds, _aimZone.bl, _aimZone.fl, _aimZone.fr) ||
265 BoundsInTriangle(bounds, _aimZone.bl, _aimZone.fr, _aimZone.br))
266 {
267 // raycast all 8 corners of the bounds, and the center of the bounds
268 bool hit = false;
269 List<Vector3> corners = new List<Vector3>();
270 corners.Add(bounds.center);
271 corners.AddRange(GetCorners(bounds));
272 foreach (Vector3 corner in corners)
273 {
274 RaycastHit hitInfo;
275 // layer mask for all layers
276 int layerMask = -1;
277 // to hit position
278 Vector3 toPosition = Vector3.Lerp(corner, bounds.center, 0.5f);
279 if (Physics.Raycast(_origin, toPosition - _origin, out hitInfo, m_range, layerMask, QueryTriggerInteraction.Ignore))
280 {
281 hitInfoR.origin = _origin;
282 hitInfoR.destination = hitInfo.point;
283
284 if (hitInfo.collider.GetComponentInParent<HealthScript>() == healthScript)
285 {
286 m_rHits.Clear();
287 m_rHits.Add(hitInfoR);
288 hit = true;
289 break;
290 }
291 }
292 }
293
294 if (hit){
295 healthScriptsInAimZone.Add(healthScript);
296 break;
297 }
298 }
299 }
300
301 // deal damage to the closest object
302 if (healthScriptsInAimZone.Count > 0)
303 {
304 //dmg
305 float calcdDamage = StatsManager.CalculateDamage(this, m_damage);
306 healthScriptsInAimZone[0].TakeDamage(calcdDamage, _owner);
307 if (m_hitEffect != null)
308 {
309 Destroy(Instantiate(
310 m_hitEffect, hitInfoR.destination,
311 Quaternion.FromToRotation(Vector3.up, hitInfoR.destination - hitInfoR.origin)),
312 2.0f);
313 }
314
315 //somatic event
316 healthScriptsInAimZone[0].GetComponentInChildren<NodeAI.NodeAI_Senses>()?.RegisterSensoryEvent(
317 _owner,
318 healthScriptsInAimZone[0].gameObject,
319 20.0f,
320 NodeAI.SensoryEvent.SenseType.SOMATIC
321 );
322 }

325 if (m_shootSound != null)
326 {
327 GameObject sound = Instantiate(m_shootSound, _origin, Quaternion.identity);
328 }
329

331 foreach (NodeAI.NodeAI_Senses sensor in FindObjectsOfType<NodeAI.NodeAI_Senses>())

349 public static bool PointInTriangle(Vector3 _point, Vector3 _triangleA, Vector3 _triangleB, Vector3 _triangleC)
350 {
351 // set y to 0
352 _point.y = 0;
353 _triangleA.y = 0;
354 _triangleB.y = 0;
355 _triangleC.y = 0;
356
357 // get vectors
358 Vector3 v0 = _triangleC - _triangleA;
359 Vector3 v1 = _triangleB - _triangleA;
360 Vector3 v2 = _point - _triangleA;
361
362 // get dot products
363 float dot00 = Vector3.Dot(v0, v0);
364 float dot01 = Vector3.Dot(v0, v1);
365 float dot02 = Vector3.Dot(v0, v2);
366 float dot11 = Vector3.Dot(v1, v1);
367 float dot12 = Vector3.Dot(v1, v2);
368
369 // get barycentric coordinates
370 float b = (dot00 * dot11 - dot01 * dot01);
371 float inv = 1 / b;
372 float u = (dot11 * dot02 - dot01 * dot12) * inv;
373 float v = (dot00 * dot12 - dot01 * dot02) * inv;
374
375 // check if point is in triangle
376 return (u >= 0) && (v >= 0) && (u + v < 1);
377 }

379 public static bool BoundsInTriangle(Bounds _bounds, Vector3 _triangleA, Vector3 _triangleB, Vector3 _triangleC)
380 {
381 //check center of bounds
382 if (PointInTriangle(_bounds.center, _triangleA, _triangleB, _triangleC)) return true;
383 //check corners of bounds
384 if (PointInTriangle(_bounds.center + new Vector3(_bounds.extents.x, 0, _bounds.extents.z), _triangleA, _triangleB, _triangleC)) return true;
385 if (PointInTriangle(_bounds.center + new Vector3(_bounds.extents.x, 0, -_bounds.extents.z), _triangleA, _triangleB, _triangleC)) return true;
386 if (PointInTriangle(_bounds.center + new Vector3(-_bounds.extents.x, 0, _bounds.extents.z), _triangleA, _triangleB, _triangleC)) return true;
387 if (PointInTriangle(_bounds.center + new Vector3(-_bounds.extents.x, 0, -_bounds.extents.z), _triangleA, _triangleB, _triangleC)) return true;
388 return false;
389 }

391 public List<Vector3> GetCorners(Bounds _bounds){
392 List<Vector3> corners = new List<Vector3>();
393 corners.Add(_bounds.center + new Vector3(_bounds.extents.x, -_bounds.extents.y, _bounds.extents.z));
394 corners.Add(_bounds.center + new Vector3(_bounds.extents.x, -_bounds.extents.y, -_bounds.extents.z));
395 corners.Add(_bounds.center + new Vector3(-_bounds.extents.x, -_bounds.extents.y, _bounds.extents.z));
396 corners.Add(_bounds.center + new Vector3(-_bounds.extents.x, -_bounds.extents.y, -_bounds.extents.z));
397 corners.Add(_bounds.center + new Vector3(_bounds.extents.x, _bounds.extents.y, _bounds.extents.z));
398 corners.Add(_bounds.center + new Vector3(_bounds.extents.x, _bounds.extents.y, -_bounds.extents.z));
399 corners.Add(_bounds.center + new Vector3(-_bounds.extents.x, _bounds.extents.y, _bounds.extents.z));
400 corners.Add(_bounds.center + new Vector3(-_bounds.extents.x, _bounds.extents.y, -_bounds.extents.z));
401 return corners;
