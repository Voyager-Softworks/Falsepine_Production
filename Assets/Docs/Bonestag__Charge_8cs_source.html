190 private Vector3 ComputeB( Vector3 circleCenter, Vector3 circleNormal, float circleRadius, Vector3 point, Vector3 direction )
191 {
192 float a = Vector3.SignedAngle( (circleCenter - point).normalized * circleRadius, direction, circleNormal );
193 float w = 0;
194 if ( a >= 0 ) w = 180 - 2 * a; // because w + a + a = 180;
195 else w = -( 180 + 2 * a );
196 Vector3 BO = Quaternion.AngleAxis(w, -circleNormal) * ((point - circleCenter).normalized * circleRadius);
197 return circleCenter + BO;
198 }
199
200 private Vector3 RandomPointOnCircle( Vector3 circleCenter, Vector3 circleNormal, float circleRadius )
201 {
202 float angle = Random.Range( 0, 360 );
203 Vector3 BO = Quaternion.AngleAxis( angle, -circleNormal ) * ((circleCenter - circleNormal * circleRadius).normalized * circleRadius);
204 return circleCenter + BO;
205 }
