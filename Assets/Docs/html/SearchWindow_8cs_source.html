54 Type[] GetInheritedClasses(Type MyType)
55 {
56 return Assembly.GetAssembly(MyType).GetTypes().Where(TheType => TheType.IsClass && !TheType.IsAbstract && TheType.IsSubclassOf(MyType)).ToArray();
57 }

62 public List<SearchTreeEntry> CreateSearchTree(SearchWindowContext context)
63 {
64 // Split inherited classes into lists based on their namespace
65 Type[] decorators = GetInheritedClasses(typeof(DecoratorBase));
66 Type[] actions = GetInheritedClasses(typeof(ActionBase));
67 Type[] conditions = GetInheritedClasses(typeof(ConditionBase));
68
69 // Create a list of all namespaces
70 var namespaces = new List<string>();
71 foreach (var decorator in decorators)
72 {
73 if (!namespaces.Contains(decorator.Namespace))
74 {
75 namespaces.Add(decorator.Namespace);
76 }
77 }
78 foreach (var action in actions)
79 {
80 if (!namespaces.Contains(action.Namespace))
81 {
82 namespaces.Add(action.Namespace);
83 }
84 }
85 foreach (var condition in conditions)
86 {
87 if (!namespaces.Contains(condition.Namespace))
88 {
89 namespaces.Add(condition.Namespace);
90 }
91 }
92
93 // Create the tree
94 var tree = new List<SearchTreeEntry>
95 {
96 new SearchTreeGroupEntry(new GUIContent("New Node"), 0),
97 new SearchTreeGroupEntry(new GUIContent("Control Node"), 1),
98 new SearchTreeEntry(new GUIContent("Sequence"))
99 {
100 userData = typeof(Sequence),
101 level = 2
102 },
103 new SearchTreeEntry(new GUIContent("Selector"))
104 {
105 userData = typeof(Selector),
106 level = 2
107 },
108 new SearchTreeEntry(new GUIContent("Parallel"))
109 {
110 userData = typeof(Parallel),
111 level = 2
112 },
113 new SearchTreeGroupEntry(new GUIContent("Decorators"), 2)
114 };
115 foreach (var n in namespaces)
116 {
117 if(n == "NodeAI")
118 {
119 foreach (var decorator in decorators)
120 {
121 if (decorator.Namespace == n)
122 {
123 tree.Add(new SearchTreeEntry(new GUIContent(decorator.Name))
124 {
125 userData = decorator, level = 3
126 });
127 }
128 }
129 }
130 else
131 {
132 tree.Add(new SearchTreeGroupEntry(new GUIContent(n ?? "Custom"), 3));
133 foreach (var decorator in decorators)
134 {
135 if (decorator.Namespace == n)
136 {
137 tree.Add(new SearchTreeEntry(new GUIContent(decorator.Name))
138 {
139 userData = decorator, level = 4
140 });
141 }
142 }
143 }
144 }
145 tree.Add(new SearchTreeGroupEntry(new GUIContent("Action"), 1));
146 foreach (var n in namespaces)
147 {
148 if(n == "NodeAI")
149 {
150 foreach (var action in actions)
151 {
152 if (action.Namespace == n)
153 {
154 tree.Add(new SearchTreeEntry(new GUIContent(action.Name))
155 {
156 userData = action, level = 2
157 });
158 }
159 }
160 }
161 else
162 {
163 tree.Add(new SearchTreeGroupEntry(new GUIContent(n ?? "Custom"), 2));
164 foreach (var action in actions)
165 {
166 if (action.Namespace == n)
167 {
168 tree.Add(new SearchTreeEntry(new GUIContent(action.Name))
169 {
170 userData = action, level = 3
171 });
172 }
173 }
174 }
175 }
176 tree.Add(new SearchTreeGroupEntry(new GUIContent("Condition"), 1));
177 foreach (var n in namespaces)
178 {
179 if(n == "NodeAI")
180 {
181 foreach (var condition in conditions)
182 {
183 if (condition.Namespace == n)

186 {

200 {

213 public bool OnSelectEntry(SearchTreeEntry entry, SearchWindowContext context)
214 {
215
216 if (((Type)entry.userData) == typeof(Sequence))
217 {
218 graphView.ContextCreateNode(selectedNode, NodeData.Type.Sequence, ((Type)entry.userData).Name, ScriptableObject.CreateInstance<Sequence>());
219
220 }
221 else if(((Type)entry.userData) == typeof(Selector))
222 {
223 graphView.ContextCreateNode(selectedNode, NodeData.Type.Selector, ((Type)entry.userData).Name, ScriptableObject.CreateInstance<Selector>());
224
225 }
226 else if(((Type)entry.userData) == typeof(Parallel))
227 {
228 graphView.ContextCreateNode(selectedNode, NodeData.Type.Parallel, ((Type)entry.userData).Name, ScriptableObject.CreateInstance<Parallel>());
229
230 }
231 else if(((Type)entry.userData).BaseType == typeof(DecoratorBase))
232 {
233 graphView.ContextCreateNode(selectedNode, NodeData.Type.Decorator, ((Type)entry.userData).Name, (DecoratorBase)ScriptableObject.CreateInstance(((Type)entry.userData)));
234
235 }
236 else if(((Type)entry.userData).BaseType == typeof(ActionBase))
237 {
238 graphView.ContextCreateNode(selectedNode, NodeData.Type.Action, ((Type)entry.userData).Name, (RuntimeBase)ScriptableObject.CreateInstance(((Type)entry.userData)));
239
240 }
241 else if(((Type)entry.userData).BaseType == typeof(ConditionBase))
242 {
243 graphView.ContextCreateNode(selectedNode, NodeData.Type.Condition, ((Type)entry.userData).Name, (RuntimeBase)ScriptableObject.CreateInstance(((Type)entry.userData)));
244 }
245 return true;
246 }
