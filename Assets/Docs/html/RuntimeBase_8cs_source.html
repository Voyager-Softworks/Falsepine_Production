509 public override NodeData.State Eval(NodeAI_Agent agent, NodeTree.Leaf current)
510 {
511
512 int successCount = 0;
513 foreach (NodeTree.Leaf child in current.children)
514 {
515 if(child.nodeData.runtimeLogic.state != NodeData.State.Running)
516 {
517 if(child.nodeData.runtimeLogic.state != NodeData.State.Idle)
518 {
519 if(child.nodeData.runtimeLogic.state == NodeData.State.Success)
520 {
521 successCount++;
522 }
523 else
524 {
525 state = NodeData.State.Failure;
526 return NodeData.State.Failure;
527 }
528 continue;
529 }
530 else
531 {
532 child.nodeData.Init(child);
533 }
534 }
535
536 switch (child.nodeData.Eval(agent, child))
537 {
538 case NodeData.State.Failure:
539 state = NodeData.State.Failure;
540 return state;
541 case NodeData.State.Success:
542 continue;
543 case NodeData.State.Running:
544 state = NodeData.State.Running;
545 return state;
546 default:
547 state = NodeData.State.Success;
548 return state;
549 }
550 }
551
552 if (successCount == current.children.Count)
553 {
554 state = NodeData.State.Success;
555 }
556 else
557 {
558 state = NodeData.State.Running;
559 }
