134 Vector3 GetClosestCoverPoint(NodeAI_Agent agent, GameObject coverFrom)
135 {
136 RaycastHit[] hits = Physics.SphereCastAll(agent.transform.position, GetProperty<float>("Distance"), agent.transform.forward, GetProperty<float>("Distance"), LayerMask.GetMask(GetProperty<string>("CoverLayer")));
137 List<Vector3> coverPoints = new List<Vector3>();
138 // For each collider, query a number of surrounding points to see if there is a navmesh edge nearby pointing away from the cover object
139 foreach(RaycastHit hit in hits)
140 {
141 Vector3 hideDirection = (hit.collider.transform.position - coverFrom.transform.position).normalized;
142 if(Vector3.Distance(hit.collider.transform.position, coverFrom.transform.position) < GetProperty<float>("Min Distance from Object") || Vector3.Distance(hit.collider.transform.position, coverFrom.transform.position) > GetProperty<float>("Max Distance from Object"))
143 {
144 continue;
145 }
146 if(hit.collider.gameObject.transform.root.gameObject.GetComponent<Cover>())
147 {
148 coverPoints.AddRange(hit.collider.gameObject.transform.root.gameObject.GetComponent<Cover>().GetAvailableCoverPoints(hideDirection));
149 }
150
151
152
153 }
154 coverPoints.Sort(delegate(Vector3 a, Vector3 b)
155 {
156 return Vector3.Distance(a, agent.transform.position).CompareTo(Vector3.Distance(b, agent.transform.position));
157 });
158 if(coverPoints.Count > 0)
159 {
160 return coverPoints[0];
161 }
162 else
163 {
164 return Vector3.zero;
165 }
