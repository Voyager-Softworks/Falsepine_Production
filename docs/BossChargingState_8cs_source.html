124 private Vector3 ComputeB( Vector3 circleCenter, Vector3 circleNormal, float circleRadius, Vector3 point, Vector3 direction )
125 {
126 float a = Vector3.SignedAngle( (circleCenter - point).normalized * circleRadius, direction, circleNormal );
127 float w = 0;
128 if ( a >= 0 ) w = 180 - 2 * a; // because w + a + a = 180;
129 else w = -( 180 + 2 * a );
130 Vector3 BO = Quaternion.AngleAxis(w, -circleNormal) * ((point - circleCenter).normalized * circleRadius);
131 return circleCenter + BO;
132 }
133
134 private Vector3 RandomPointOnCircle( Vector3 circleCenter, Vector3 circleNormal, float circleRadius )
135 {
136 float angle = Random.Range( 0, 360 );
137 Vector3 BO = Quaternion.AngleAxis( angle, -circleNormal ) * ((circleCenter - circleNormal * circleRadius).normalized * circleRadius);
138 return circleCenter + BO;
139 }
