92 public void AddProperty<T>(string name, T initialValue, bool isOutput) //Adds a property
93 {
94 if(properties == null)
95 {
96 properties = new List<NodeData.SerializableProperty>();
97 }
98 foreach (NodeData.SerializableProperty property in properties)
99 {
100 if (property.name == name.ToUpper())
101 {
102 Debug.LogError("Property with name " + name + " already exists");
103 return;
104 }
105 }
106 NodeData.SerializableProperty newProp = new NodeData.SerializableProperty()
107 {
108 name = name.ToUpper(),
109 serializedTypename = typeof(T).AssemblyQualifiedName,
110 output = isOutput,
111 };
112 switch(newProp.type.Name)
113 {
114 case "Int32":
115 newProp.ivalue = (int)(object)initialValue;
116 break;
117 case "Single":
118 newProp.fvalue = (float)(object)initialValue;
119 break;
120 case "Boolean":
121 newProp.bvalue = (bool)(object)initialValue;
122 break;
123 case "String":
124 newProp.svalue = (string)(object)initialValue;
125 break;
126 case "Vector2":
127 newProp.v2value = (Vector2)(object)initialValue;
128 break;
129 case "Vector3":
130 newProp.v3value = (Vector3)(object)initialValue;
131 break;
132 case "Vector4":
133 newProp.v4value = (Vector4)(object)initialValue;
134 break;
135 case "Color":
136 newProp.cvalue = (Color)(object)initialValue;
137 break;
138 default:
139 newProp.ovalue = (Object)(object)initialValue;
140 break;
141 }
142
143 properties.Add(newProp);
144 }

301 public T GetProperty<T>(string name)
302 {
303 if(properties == null)

307 foreach (NodeData.SerializableProperty property in properties)

309 if (property.name == name.ToUpper() && property.serializedTypename == typeof(T).AssemblyQualifiedName)
310 {
311
312 switch(property.type.Name)
313 {
314 case "Int32":
315 return (T)(object)property.ivalue;
316
317 case "Single":
318 return (T)(object)property.fvalue;
319
320 case "Boolean":
321 return (T)(object)property.bvalue;
322
323 case "String":
324 return (T)(object)property.svalue;
325
326 case "Vector2":

328
329 case "Vector3":

331

333 return (T)(object)property.v4value;
334
