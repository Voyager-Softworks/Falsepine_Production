124 isAiming = rangedWeapon.m_isAiming;

126 }
127

130 moveDir *= walkSpeed;
131 }
132 else if(sprintAction.ReadValue<float>() > 0.1f)

134 moveDir *= sprintSpeed;

137 {
138 moveDir *= jogSpeed;
139 }

141 moveDir.y = -2.0f;

143 animVelocity = Vector3.Lerp(animVelocity, moveDir, Time.deltaTime * 10f);
144 if(playerHealth.isStunned) return;
145 //check if placing bear trap anim is playing
146 if (_animator.GetCurrentAnimatorStateInfo(0).IsName("Player_Rig_BearTrap_Revolver") ||
147 _animator.GetCurrentAnimatorStateInfo(0).IsName("Player_Rig_BearTrap_Rifle") ||

153 //roll
154 if (rollTimer > 0)

156 rollTimer -= Time.deltaTime;

174 rollTimer = 0;

294 public Vector3 GetMouseAimPoint(){
295 //mouse raycast to get direction
296 Ray ray = Camera.main.ScreenPointToRay(Mouse.current.position.ReadValue());
297 PlayerInventoryInterface pii = GetComponent<PlayerInventoryInterface>();
298 if (!pii) return mouseAimPoint;
299 GameObject weaponFirepoint = pii.GetWeaponFirepoint(pii.selectedWeapon);
300 if (!weaponFirepoint) return mouseAimPoint;
301 //find where ray intersects on the plane of the player
302 Plane playerPlane = new Plane(Vector3.up, weaponFirepoint.transform.position);
303 float rayDistance;
304 if (playerPlane.Raycast(ray, out rayDistance)){
305 //get mouse hit pos
306 Vector3 hitPoint = ray.GetPoint(rayDistance);
307 mouseAimPoint = hitPoint;
308 }
309 return mouseAimPoint;
310 }
