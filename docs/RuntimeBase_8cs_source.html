58 public void AddProperty<T>(string name, T initialValue)
59 {
60 if(properties == null)
61 {
62 properties = new List<NodeData.SerializableProperty>();
63 }
64 foreach (NodeData.SerializableProperty property in properties)
65 {
66 if (property.name == name.ToUpper())
67 {
68 Debug.LogError("Property with name " + name + " already exists");
69 return;
70 }
71 }
72 NodeData.SerializableProperty newProp = new NodeData.SerializableProperty()
73 {
74 name = name.ToUpper(),
75 serializedTypename = typeof(T).AssemblyQualifiedName,
76 };
77 switch(newProp.type.Name)
78 {
79 case "Int32":
80 newProp.ivalue = (int)(object)initialValue;
81 break;
82 case "Single":
83 newProp.fvalue = (float)(object)initialValue;
84 break;
85 case "Boolean":
86 newProp.bvalue = (bool)(object)initialValue;
87 break;
88 case "String":
89 newProp.svalue = (string)(object)initialValue;
90 break;
91 case "Vector2":
92 newProp.v2value = (Vector2)(object)initialValue;
93 break;
94 case "Vector3":
95 newProp.v3value = (Vector3)(object)initialValue;
96 break;
97 case "Vector4":
98 newProp.v4value = (Vector4)(object)initialValue;
99 break;
100 case "Color":
101 newProp.cvalue = (Color)(object)initialValue;
102 break;
103 default:
104 newProp.ovalue = (Object)(object)initialValue;
105 break;
106 }
107
108 properties.Add(newProp);
109 }

820 public Sequence()
821 {
822 tooltip = "Runs all children in order (top to bottom), until one fails, or all have succeeded.";
823 }
824 public override NodeData.State Eval(NodeAI_Agent agent, NodeTree.Leaf current)
825 {
826
827 int successCount = 0;
828 foreach (NodeTree.Leaf child in current.children)
829 {
830 if(child.nodeData.runtimeLogic.state != NodeData.State.Running)
831 {
832 if(child.nodeData.runtimeLogic.state != NodeData.State.Idle)
833 {
834 if(child.nodeData.runtimeLogic.state == NodeData.State.Success)
835 {
836 successCount++;
837 }
838 else
839 {
840 state = NodeData.State.Failure;
841 return NodeData.State.Failure;
842 }
843 continue;
844 }
845 else
846 {
847 child.nodeData.Init(child);
848 }
849 }
850
851 switch (child.nodeData.Eval(agent, child))
852 {
853 case NodeData.State.Failure:
854 state = NodeData.State.Failure;
855 return state;
856 case NodeData.State.Success:
857 continue;
858 case NodeData.State.Running:
859 state = NodeData.State.Running;
860 return state;
861 default:
862 state = NodeData.State.Success;
863 return state;
864 }
865 }
866
867 if (successCount == current.children.Count)
868 {
869 state = NodeData.State.Success;
870 }
871 else
872 {
873 state = NodeData.State.Running;
874 }
875 return state;
876 }

888 public Parallel()
889 {
890 tooltip = "Runs all children simultaneously, until all have failed or one has succeeded.";
891 }
892 public override NodeData.State Eval(NodeAI_Agent agent, NodeTree.Leaf current)
893 {
894 int failCount = 0;
895 bool success = false;
896 foreach (NodeTree.Leaf child in current.children)
897 {
898 if(child.nodeData.runtimeLogic.state == NodeData.State.Idle) {
899 child.nodeData.Init(child);
900 }
901 switch (child.nodeData.Eval(agent, child))
902 {
903 case NodeData.State.Failure:
904 failCount++;
905 continue;
906 case NodeData.State.Success:
907 success = true;
908 continue;
909 case NodeData.State.Running:
910 state = NodeData.State.Running;
911 continue;
912 default:
913 state = NodeData.State.Success;
914 return state;
915 }
916 }
917 if (failCount == current.children.Count)
918 {
919 state = NodeData.State.Failure;
920 }
921 else if(success)
922 {
923 state = NodeData.State.Success;
924 }
925 else
926 {
927 state = NodeData.State.Running;
928 }
929 return state;
