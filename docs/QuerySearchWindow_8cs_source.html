58 public void PopulateSearchTreeRecursively(List<SearchTreeEntry> searchTree, NamespaceParser.NamespaceEntry entry, int depth = 0)
59 {
60 if(entry.Namespace == "" || entry.Namespace == "NodeAI") depth = 0;
61 else searchTree.Add(new SearchTreeGroupEntry(new GUIContent(entry.Namespace), depth));
62 foreach (var subnamespace in entry.Subnamespaces)
63 {
64 PopulateSearchTreeRecursively(searchTree, subnamespace, depth + 1);
65 }
66
67 foreach (var type in entry.Classes)
68 {
69 searchTree.Add(new SearchTreeEntry(new GUIContent(type.Name))
70 {
71 userData = type, level = depth + 1
72 });
73 }
74 }
80 public List<SearchTreeEntry> CreateSearchTree(SearchWindowContext context)
81 {
82 // Split inherited classes into lists based on their namespace
83 Type[] queries = GetInheritedClasses(typeof(Query));
84
85 var parsedTree = NamespaceParser.TreeFromTypes(queries);
86
87 var tree = new List<SearchTreeEntry>
88 {
89 new SearchTreeGroupEntry(new GUIContent("New Query"), 0),
90 };
91
92 PopulateSearchTreeRecursively(tree, parsedTree, 0);
93
94 return tree;
