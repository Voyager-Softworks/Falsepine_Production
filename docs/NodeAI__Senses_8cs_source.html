281 public GameObject[] GetVisibleObjects()
282 {
283 RaycastHit[] hits = Physics.SphereCastAll(eyesBone.position, sightDistance/2, eyesForward, sightDistance/2, sightMask);
284 List<GameObject> visibleObjects = new List<GameObject>();
285 foreach(RaycastHit hit in hits)
286 {
287 if(hit.collider.gameObject == this.gameObject)
288 {
289 continue;
290 }
291 if(hit.distance > sightDistance)
292 {
293 continue;
294 }
295 Vector3 dir = hit.transform.position - transform.position;
296 if(Vector3.Angle(eyesForward, dir) < sightAngle / 2)
297 {
298 //check line of sight
299 RaycastHit[] hits2 = Physics.RaycastAll(eyesBone.position, dir, dir.magnitude);
300 bool lineOfSight = true;
301 foreach(RaycastHit hit2 in hits2)
302 {
303 if(hit.collider.gameObject == this.gameObject)
304 {
305 continue;
306 }
307 if(hit2.collider.gameObject != hit.collider.gameObject)
308 {
309 lineOfSight = false;
310 break;
311 }
312 }
313 if(lineOfSight)
314 {
315 visibleObjects.Add(hit.collider.gameObject);
316 }
317 }
318 }
319 return visibleObjects.ToArray();
320 }
