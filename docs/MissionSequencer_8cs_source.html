59 void GenerateNewSequence(Area area)
60 {
61 currentSequence = new Queue<string>();
62 // Pick a random sequence from the list of sequences, where the area matches the parameter.
63 MissionSequence sequence = missionSequences.Where(x => x.area == area).OrderBy(x => Random.value).First();
64 // Add the start scene to the sequence.
65 currentSequence.Enqueue(sequence.missionStartScene);
66 // Add the scenes in random order to the sequence.
67 sequence.scenes.OrderBy(x => Random.value).ToList().ForEach(x => currentSequence.Enqueue(x));
68 // Add the end scene to the sequence.
69 currentSequence.Enqueue(sequence.missionEndScene);
70 }
