33 public override NodeData.State Eval(NodeAI_Agent agent, NodeTree.Leaf current)
34 {
35 if(navAgent == null)
36 {
37 navAgent = agent.GetComponent<NavMeshAgent>();
38 if(navAgent == null)
39 {
40 Debug.LogError("SeekClosestWithTag: NavMeshAgent not found on agent");
41 return NodeData.State.Failure;
42 }
43 }
44 var tag = GetProperty<string>("Tag");
45 var range = GetProperty<float>("Range");
46
47 GameObject[] targets = GameObject.FindGameObjectsWithTag(tag);
48 if(targets.Length <= 0)
49 {
50
51 return NodeData.State.Failure;
52 }
53 var closest = targets[0];
54
55 foreach (var target in targets)
56 {
57 var distance = Vector3.Distance(agent.transform.position, target.transform.position);
58 if (distance < Vector3.Distance(agent.transform.position, closest.transform.position))
59 {
60 closest = target;
61 }
62 }
63 if(Vector3.Distance(agent.transform.position, closest.transform.position) < range)
64 {
65 navAgent.SetDestination(closest.transform.position);
66 if(navAgent.stoppingDistance > Vector3.Distance(agent.transform.position, closest.transform.position))
67 {
68 navAgent.isStopped = true;
69 return NodeData.State.Success;
70 }
71 else
72 {
73 navAgent.isStopped = false;
74 return NodeData.State.Running;
75 }
76
77 }
78 return NodeData.State.Failure;
79 }

81 public override void DrawGizmos(NodeAI_Agent agent)
82 {
83 if(state != NodeData.State.Running)
84 {
85 return;
86 }
87 var tag = GetProperty<string>("Tag");
88 var range = GetProperty<float>("Range");
89 var targets = GameObject.FindGameObjectsWithTag(tag);
90 foreach (var target in targets)
91 {
92 var distance = Vector3.Distance(agent.transform.position, target.transform.position);
93 if (distance < range)
94 {
95 Gizmos.color = Color.green;
96 Gizmos.DrawLine(agent.transform.position, target.transform.position);
97 Gizmos.DrawIcon(target.transform.position, "Download-Available", true, Color.green);
98 }
99 }
