89 public static NodeTree CreateFromNodeData(NodeData rootNodeData, List<NodeData> data)
90 {
91 NodeTree nodeTree = new NodeTree();
92 nodeTree.rootNode = rootNodeData;
93 nodeTree.nodes = data;
94 nodeTree.rootLeaf = new Leaf();
95 nodeTree.rootLeaf.nodeData = rootNodeData;
96 nodeTree.BuildTree(nodeTree.rootLeaf);
97 return nodeTree;
98 }

100 private void BuildTree(Leaf leaf)
101 {
102 foreach (var child in leaf.nodeData.childGUIDs)
103 {
104 var childLeaf = new Leaf();
105 childLeaf.nodeData = nodes.Find(x => x.GUID == child);
106 leaf.children.Add(childLeaf);
107 leaf.children.Sort((x, y) => x.nodeData.position.y.CompareTo(y.nodeData.position.y));
108 BuildTree(childLeaf);
109 }
