54 public void Serialize(NodeAI_Behaviour nodeAI_Behaviour)
55 {
56 nodeAI_Behaviour.nodeData = new List<NodeData>(); // Initialize the node data list
57 nodeAI_Behaviour.queries = new List<Query>(); // Initialize the query list
58
59 // Serialize the nodes
60 foreach (var node in nodes)
61 {
62 // Create a new node data
63 var nodeData = new NodeData
64 {
65 GUID = node.GUID, // Set the GUID
66 nodeType = node.nodeType, // Set the node type
67 position = node.GetPosition().position, // Set the position
68 childGUIDs = new List<string>(), // Initialize the child GUID list
69 title = node.title, // Set the title
70 runtimeLogic = (node.runtimeLogic ?? null), // Set the runtime logic
71 query = (node.query ?? null) // Set the query
72 };
73
74 // Setup ScriptableObject components if they are not null
75 if(nodeData.runtimeLogic != null)
76 {
77 nodeData.runtimeLogic.state = NodeData.State.Idle;
78 }
79 if(nodeData.query != null)
80 {
81 nodeAI_Behaviour.queries.Add(nodeData.query);
82 }
83
84 // Store the parameter reference
85 if(node.nodeType == NodeData.Type.Parameter) nodeData.parentGUID = node.paramReference;
86
87 // Serialise connections
88 foreach (var input in node.inputPorts)
89 {
90 if(nodeData.runtimeLogic != null) //If this node has runtime logic
91 {
92 if (input.connections.Count() > 0) //If this node is connected to anything
93 {
94 if(((Node)input.connections.First().output.node).nodeType == NodeData.Type.Parameter) // If the connected node is a parameter
95 {
96 // Set the properties parameter reference as the nodes' parameter reference
97 nodeData.runtimeLogic.SetPropertyParamReference(input.portName, ((Node)input.connections.First().output.node).paramReference);
98 }
99 else
100 {
101 // Set the properties parameter reference as the nodes' GUID
102 nodeData.runtimeLogic.SetPropertyParamReference(input.portName, ((Node)input.connections.First().output.node).query.GetProperties().Find(x => x.name == input.connections.First().output.portName).GUID);
103 }
104 nodeData.runtimeLogic.SetPropertyGUID(input.portName, ((Node)input.connections.First().output.node).GUID); // Set the properties GUID
105 }
106 else
107 {
108 nodeData.runtimeLogic.SetPropertyParamReference(input.portName, "null"); // Set the properties parameter reference to null
109 nodeData.runtimeLogic.SetPropertyGUID(input.portName, "null"); // Set the properties GUID to null
110 }
111 }
112 else if(nodeData.query != null)
113 {
114 if (input.connections.Count() > 0)
115 {
116 if(((Node)input.connections.First().output.node).nodeType == NodeData.Type.Parameter) //If a parameter is connected
117 {
118 // Set the properties parameter reference as the nodes' parameter reference
119 nodeData.query.SetPropertyParamReference(input.portName, ((Node)input.connections.First().output.node).paramReference);
120 }
121 else
122 {
123 // Set the properties parameter reference as the nodes' GUID
124 nodeData.query.SetPropertyParamReference(input.portName, ((Node)input.connections.First().output.node).query.GetProperties().Find(x => x.name == input.connections.First().output.portName).GUID);
125 }
126 // Set the properties GUID
127 nodeData.query.SetPropertyGUID(input.portName, ((Node)input.connections.First().output.node).GUID);
128 }
129 else
130 {
131 nodeData.query.SetPropertyParamReference(input.portName, "null"); // Set the properties parameter reference to null
132 nodeData.query.SetPropertyGUID(input.portName, "null"); // Set the properties GUID to null
133 }
134
135 }
136 }
137 nodeAI_Behaviour.nodeData.Add(nodeData); // Add the node data to the list
138 }
139
140 foreach (var edge in edges)
141 {
142 if(((Node)edge.output.node).nodeType == NodeData.Type.Parameter) //Parameter specific logic
143 {
144 ((Node)edge.input.node).runtimeLogic?.SetPropertyGUID(edge.input.portName, ((Node)edge.output.node).GUID);
145 ((Node)edge.input.node).query?.SetPropertyGUID(edge.input.portName, ((Node)edge.output.node).GUID);
146 }
147 else if(((Node)edge.output.node).nodeType == NodeData.Type.Query) //Query specific logic
148 {
149 ((Node)edge.input.node).runtimeLogic?.SetPropertyParamReference(edge.input.portName, ((Node)edge.output.node).query.GetProperties().Find(x => x.name == edge.output.portName).GUID);
150 ((Node)edge.input.node).query?.SetPropertyParamReference(edge.input.portName, ((Node)edge.output.node).query.GetProperties().Find(x => x.name == edge.output.portName).GUID);
151 ((Node)edge.input.node).runtimeLogic?.SetPropertyGUID(edge.input.portName, ((Node)edge.output.node).GUID);
152 ((Node)edge.input.node).query?.SetPropertyGUID(edge.input.portName, ((Node)edge.output.node).GUID);
153 }
154 else
155 {
156 var inputNodeData = nodeAI_Behaviour.nodeData.Find(x => x.GUID == ((Node)edge.input.node).GUID);
157 var outputNodeData = nodeAI_Behaviour.nodeData.Find(x => x.GUID == ((Node)edge.output.node).GUID);
158
159 inputNodeData.parentGUID = outputNodeData.GUID;
160 outputNodeData.childGUIDs.Add(inputNodeData.GUID);
161 }
162 }
163
164
165 //Repopulate exposed properties
166 nodeAI_Behaviour.exposedProperties.Clear();
167 foreach(var p in target.exposedProperties)
168 {
169 nodeAI_Behaviour.exposedProperties.Add(p);
170 }
171
172 //Serialise Groups
173 if(nodeAI_Behaviour.nodeGroups == null) nodeAI_Behaviour.nodeGroups = new List<NodeData.NodeGroup>();
174 nodeAI_Behaviour.nodeGroups.Clear();
175 foreach(var g in target.graphElements.ToList().OfType<Group>())
176 {
177 NodeData.NodeGroup nodeGroup = new NodeData.NodeGroup
178 {
179 title = g.title,
180 childGUIDs = new List<string>()
181 };
182 foreach(var n in g.containedElements.ToList().OfType<Node>())
183 {
184 nodeGroup.childGUIDs.Add(n.GUID);
185 }
186 nodeAI_Behaviour.nodeGroups.Add(nodeGroup);

196 public void Deserialize(NodeAI_Behaviour nodeAI_Behaviour)
197 {
198 // Clear the graph
199 target.exposedProperties.Clear();
200 target.blackboard.Clear();
201
202 // Deserialise exposed properties
203 foreach (var property in nodeAI_Behaviour.exposedProperties)
204 {
205 target.AddPropertyToBlackboard(property);
206 }
207
208 // Clear nodes and edges
209 foreach (var node in nodes)
210 {
211 target.RemoveElement(node);
212 }
213 foreach (var edge in edges)
214 {
215 target.RemoveElement(edge);
216 }
217
218 // Deserialise nodes
219 foreach (var nodeData in nodeAI_Behaviour.nodeData)
220 {
221 var node = target.GenerateNode(nodeData);
222 target.AddElement(node);
223 nodes.Add(node);
224 }
225
226 // Deserialise edges
227 foreach (var nodeData in nodeAI_Behaviour.nodeData)
228 {
229 var node = nodes.Find(x => x.GUID == nodeData.GUID);
230
231 // If the node can have sequence inputs, then apply them
232 if (nodeData.nodeType != NodeData.Type.EntryPoint &&
233 nodeData.nodeType != NodeData.Type.Parameter &&
234 nodeData.nodeType != NodeData.Type.Query)
235 {
236 var parent = nodes.Find(x => x.GUID == nodeData.parentGUID);
237 if(parent != null)
238 {
239 var edge = parent.outputPort.ConnectTo(node.inputPort);
240 edges.Add(edge);
241 target.AddElement(edge);
242 }
243
244 }
245 else if(nodeData.nodeType == NodeData.Type.Parameter) // If the node is a parameter then deserialize its parameter reference
246 {
247 node.paramReference = nodeData.parentGUID;
248
249 }
250 foreach(Port input in node.inputPorts) //Reconnect edges
251 {
252 if(nodeData.runtimeLogic != null)
253 {
254 NodeData.Property property = nodeData.runtimeLogic.GetProperties().Find(x => x.name == input.portName);
255 string connGUID = property.GUID;
256 var conn = nodes.Find(x => x.GUID == connGUID);
257 if (conn != null)
258 {
259 if(conn.nodeType == NodeData.Type.Query)
260 {
261 var outputPort = conn.outputPorts.Find(x => x.portName == conn.query.GetProperties().Find(y => y.GUID == property.paramReference).name);
262 var edge = outputPort.ConnectTo(input);
263 edges.Add(edge);
264 target.AddElement(edge);
265 }
266 else
267 {
268 var edge = conn.outputPort.ConnectTo(input);
269 edges.Add(edge);
270 target.AddElement(edge);
271 }
272
273 }
274 }
275 else if(nodeData.query != null)
276 {
277 NodeData.Property property = nodeData.query.GetProperties().Find(x => x.name == input.portName);
278 string connGUID = property.GUID;
279 var conn = nodes.Find(x => x.GUID == connGUID);
280 if (conn != null)
281 {
282 if(conn.nodeType == NodeData.Type.Query)
283 {
284 var outputPort = conn.outputPorts.Find(x => x.portName == conn.query.GetProperties().Find(y => y.GUID == property.paramReference).name);
285 var edge = outputPort.ConnectTo(input);
286 edges.Add(edge);
287 target.AddElement(edge);
288 }
289 else
290 {
291 var edge = conn.outputPort.ConnectTo(input);
292 edges.Add(edge);
293 target.AddElement(edge);
294 }
295 }
296 }
297 }
298
299 }
300 target.graphElements.ToList().OfType<Group>().ToList().ForEach(x => target.RemoveElement(x));
301 foreach (var nodeGroup in nodeAI_Behaviour.nodeGroups)
302 {
303 List<Node> nodesToAdd = new List<Node>();
304 foreach (var nodeGUID in nodeGroup.childGUIDs)
305 {
306 var node = nodes.Find(x => x.GUID == nodeGUID);
307 nodesToAdd.Add(node);
308 }
309
310 target.CreateGroup(nodeGroup.title, nodesToAdd);
311
312 }
