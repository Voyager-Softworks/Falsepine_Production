57public class EditorInputDialog : EditorWindow
58{

67 #region OnGUI()
68 void OnGUI()
69 {
70 // Check if Esc/Return have been pressed
71 var e = Event.current;
72 if( e.type == EventType.KeyDown )
73 {
74 switch( e.keyCode )
75 {
76 // Escape pressed
77 case KeyCode.Escape:
78 shouldClose = true;
79 e.Use();
80 break;
81
82 // Enter pressed
83 case KeyCode.Return:
84 case KeyCode.KeypadEnter:
85 onOKButton?.Invoke();
86 shouldClose = true;
87 e.Use();
88 break;
89 }
90 }
91
92 if( shouldClose ) { // Close this dialog
93 Close();
94 //return;
95 }
96
97 // Draw our control
98 var rect = EditorGUILayout.BeginVertical();
99
100 EditorGUILayout.Space( 12 );
101 EditorGUILayout.LabelField( description );
102
103 EditorGUILayout.Space( 8 );
104 GUI.SetNextControlName( "inText" );
105 inputText = EditorGUILayout.TextField( "", inputText );
106 GUI.FocusControl( "inText" ); // Focus text field
107 EditorGUILayout.Space( 12 );
108
109 // Draw OK / Cancel buttons
110 var r = EditorGUILayout.GetControlRect();
111 r.width /= 2;
112 if( GUI.Button( r, okButton ) ) {
113 onOKButton?.Invoke();
114 shouldClose = true;
115 }
116
117 r.x += r.width;
118 if( GUI.Button( r, cancelButton ) ) {
119 inputText = null; // Cancel - delete inputText
120 shouldClose = true;
121 }
122
123 EditorGUILayout.Space( 8 );
124 EditorGUILayout.EndVertical();
125
126 // Force change size of the window
127 if( rect.width != 0 && minSize != rect.size ) {
128 minSize = maxSize = rect.size;
129 }
130
131 // Set dialog position next to mouse position
132 if( !initializedPosition && e.type == EventType.Layout )
133 {
134 initializedPosition = true;
135
136 // Move window to a new position. Make sure we're inside visible window
137 var mousePos = GUIUtility.GUIToScreenPoint( Event.current.mousePosition );
138 mousePos.x += 32;
139 if( mousePos.x + position.width > maxScreenPos.x ) mousePos.x -= position.width + 64; // Display on left side of mouse
140 if( mousePos.y + position.height > maxScreenPos.y ) mousePos.y = maxScreenPos.y - position.height;
141
142 position = new Rect( mousePos.x, mousePos.y, position.width, position.height );
143
144 // Focus current window
145 Focus();
146 }
147 }
148 #endregion OnGUI()
