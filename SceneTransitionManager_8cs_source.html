132 Vector2 RotatePoint(Vector2 point, float angle)
133 {
134 //rotate the point around the origin
135 float radians = angle * Mathf.Deg2Rad;
136 float sin = Mathf.Sin(radians);
137 float cos = Mathf.Cos(radians);
138 float tx = point.x;
139 float ty = point.y;
140 point.x = (cos * tx) - (sin * ty);
141 point.y = (sin * tx) + (cos * ty);
142 return point;
